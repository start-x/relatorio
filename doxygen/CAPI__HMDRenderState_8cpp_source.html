53 ovrHmdDesc HMDRenderState::GetDesc()
54 {
55  ovrHmdDesc d;
56  memset(&d, 0, sizeof(d));
57 
58  d.Type = ovrHmd_Other;
59 
60  d.ProductName = HMDInfo.ProductName;
61  d.Manufacturer = HMDInfo.Manufacturer;
62  d.Resolution.w = HMDInfo.ResolutionInPixels.w;
63  d.Resolution.h = HMDInfo.ResolutionInPixels.h;
64  d.WindowsPos.x = HMDInfo.DesktopX;
65  d.WindowsPos.y = HMDInfo.DesktopY;
66  d.DisplayDeviceName = HMDInfo.DisplayDeviceName;
67  d.DisplayId = HMDInfo.DisplayId;
68 
69  d.HmdCaps = ovrHmdCap_Present | ovrHmdCap_NoVSync;
70  d.SensorCaps = ovrSensorCap_YawCorrection | ovrSensorCap_Orientation;
71  d.DistortionCaps = ovrDistortionCap_Chromatic | ovrDistortionCap_TimeWarp | ovrDistortionCap_Vignette;
72 
73  if (strstr(HMDInfo.ProductName, "DK1"))
74  {
75  d.Type = ovrHmd_DK1;
76  }
77  else if (strstr(HMDInfo.ProductName, "DK2"))
78  {
79  d.Type = ovrHmd_DK2;
80  d.HmdCaps |= ovrHmdCap_LowPersistence |
81  ovrHmdCap_LatencyTest | ovrHmdCap_DynamicPrediction;
82  d.SensorCaps |= ovrSensorCap_Position;
83  }
84 
85  DistortionRenderDesc& leftDistortion = Distortion[0];
86  DistortionRenderDesc& rightDistortion = Distortion[1];
87 
88  // The suggested FOV (assuming eye rotation)
89  d.DefaultEyeFov[0] = CalculateFovFromHmdInfo(StereoEye_Left, leftDistortion, RenderInfo, OVR_DEFAULT_EXTRA_EYE_ROTATION);
90  d.DefaultEyeFov[1] = CalculateFovFromHmdInfo(StereoEye_Right, rightDistortion, RenderInfo, OVR_DEFAULT_EXTRA_EYE_ROTATION);
91 
92  // FOV extended across the entire screen
93  d.MaxEyeFov[0] = GetPhysicalScreenFov(StereoEye_Left, leftDistortion);
94  d.MaxEyeFov[1] = GetPhysicalScreenFov(StereoEye_Right, rightDistortion);
95 
96  if (HMDInfo.Shutter.Type == HmdShutter_RollingRightToLeft)
97  {
98  d.EyeRenderOrder[0] = ovrEye_Right;
99  d.EyeRenderOrder[1] = ovrEye_Left;
100  }
101  else
102  {
103  d.EyeRenderOrder[0] = ovrEye_Left;
104  d.EyeRenderOrder[1] = ovrEye_Right;
105  }
106 
107  return d;
108 }
