  71 void Ovr::getXYZW(float * x, float * y, float * z, float * w)

  75 
  76         int attempts = 10000;
  77 
  78         while(attempts && !(ss.StatusFlags & (ovrStatus_OrientationTracked | ovrStatus_PositionTracked)))
  79                 attempts--;
  80 
  81         if(attempts)
  82         {
  83                 ovrPosef pose = ss.Predicted.Pose;
  84 
  85                 ovrQuatf orientation = pose.Orientation;
  86         ovrVector3f position = pose.Position;
  87 
  88                 *x = orientation.x;
  89                 *y = orientation.y;
  90                 *z = orientation.z;
  91                 *w = orientation.w;
  92
