  45 
  46     if ((displayInfo.DistortionType & SensorDisplayInfoImpl::Mask_BaseFmt) == SensorDisplayInfoImpl::Base_Distortion)
  47     {

  51 
  52     visitor.Visit(hmdCreateDesc);
