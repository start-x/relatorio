  83 wchar_t* OVR_CDECL OVR_wcsncpy(wchar_t* dest, UPInt destsize, const wchar_t* src, UPInt count)
  84 {
  85 #if defined(OVR_MSVC_SAFESTRING)
  86     wcsncpy_s(dest, destsize, src, count);
  87     return dest;
  88 #else
  89     UPInt srclen = OVR_wcslen(src);
  90     UPInt l = Alg::Min(srclen, count);
  91     l = (l < destsize) ? l : destsize;
  92     memcpy(dest, src, l * sizeof(wchar_t));
  93     if (count > srclen)
  94     {
  95         UPInt remLen = Alg::Min(destsize - l, (count - srclen));
  96         memset(&dest[l], 0, sizeof(wchar_t)*remLen);
  97     }
  98     else if (l < destsize)
  99         dest[l] = 0;
 100     return dest;
 101 #endif
 102 }

 107 #if defined(OVR_MSVC_SAFESTRING)
 108     wcscat_s(dest, destsize, src);
 109     return dest;

 111     OVR_UNUSED(destsize);
 112     wcscat(dest, src);
 113     return dest;
 114 #else
 115     UPInt dstlen = OVR_wcslen(dest); // do not incl term null
 116     UPInt srclen = OVR_wcslen(src) + 1; // incl term null
 117     UPInt copylen = (dstlen + srclen < destsize) ? srclen : destsize - dstlen;
 118     memcpy(dest + dstlen, src, copylen * sizeof(wchar_t));
 119     return dest;
 120 #endif

 135 int OVR_CDECL OVR_wcscmp(const wchar_t* a, const wchar_t* b)
 136 {
 137 #if defined(OVR_OS_WIN32) || defined(OVR_OS_LINUX)
 138     return wcscmp(a, b);
 139 #else
 140     // not supported, use custom implementation
 141     const wchar_t *pa = a, *pb = b;
 142     while (*pa && *pb)
 143     {
 144         wchar_t ca = *pa;
 145         wchar_t cb = *pb;
 146         if (ca < cb)
 147             return -1;
 148         else if (ca > cb)
 149             return 1;
 150         pa++;
 151         pb++;
 152     }
 153     if (*pa)
 154         return 1;
 155     else if (*pb)
 156         return -1;
 157     else
 158         return 0;
 159 #endif
 160 }

 169 #endif
 170 #elif defined(OVR_OS_MAC) || defined(__CYGWIN__) || defined(OVR_OS_ANDROID) || defined(OVR_OS_IPHONE)
 171     // not supported, use custom implementation
 172     const wchar_t *pa = a, *pb = b;
 173     while (*pa && *pb)
 174     {
 175         wchar_t ca = OVR_towlower(*pa);
 176         wchar_t cb = OVR_towlower(*pb);
 177         if (ca < cb)
 178             return -1;
 179         else if (ca > cb)
 180             return 1;
 181         pa++;
 182         pb++;
 183     }
 184     if (*pa)
 185         return 1;
 186     else if (*pb)
 187         return -1;
 188     else
 189         return 0;
 190 #else
 191     return wcscasecmp(a, b);
 192 #endif
 193 }

 197 {
 198 #if !defined(OVR_OS_ANDROID)
 199     const char s = *localeconv()->decimal_point;
 200 
 201     if (s != '.')
 202     {
 203         char buffer[347 + 1];
 204 
 205         OVR_strcpy(buffer, sizeof(buffer), string);
 206 
 207         for (char* c = buffer; *c != '\0'; ++c)
 208         {
 209             if (*c == '.')
 210             {
 211                 *c = s;
 212                 break;
 213             }
 214         }
 215 
 216         return strtod(buffer, tailptr);
 217     }
 218 #endif
 219 
 220     return strtod(string, tailptr);
