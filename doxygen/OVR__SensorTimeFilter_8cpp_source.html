53 // Ideally, the following would always be true:
54 // - NewSampleTime > PrevSample
55 // - NewSampleTime < now systemTime
56 // - (NewSampleTime - PrevSampleTime) == integration delta, matching
57 // HW sample time difference + drift
58 //
59 // In practice, these issues affect us:
60 // - System thread can be suspended for a while
61 // - System de-buffering of recorded samples cause deviceTime to advance up
62 // much faster then system time for ~100+ samples
63 // - Device (DK1) and system clock granularities are high; this can
64 // lead to potentially having estimations in the future
65 //

94 double SensorTimeFilter::SampleToSystemTime(double sampleDeviceTime, double systemTime,
95  double prevResult, const char* debugTag)
96 {
97  double clockDelta = systemTime - sampleDeviceTime + FilterSettings.ClockDeltaAdjust;
98  double deviceTimeDelta = sampleDeviceTime - LastLargestDeviceTime;
99  double result;
100 
101  // Collect a sample ClockDelta for a "MinimumWindow" or process
102  // the window by adjusting drift rates if it's full of samples.
103  // - (deviceTimeDelta < 1.0f) is a corner cases, as it would imply timestamp skip/wrap.
104 
105  if (ClockInitialized)
106  {
107  // Samples in the past commonly occur if they come from separately incrementing
108  // data channels. Just adjust them with ClockDelta.
109 
110  if (deviceTimeDelta < 0.0)
111  {
112  result = sampleDeviceTime + ClockDelta;
113 
114  if (result > (prevResult - 0.00001))
115  goto clamp_and_log_result;
116 
117  // Consistent samples less then prevResult for indicate a back-jump or bad input.
118  // In this case we return prevResult for a while, then reset filter if it keeps going.
119  if (PastSampleResetTime < 0.0001)
120  {
121  PastSampleResetTime = systemTime + FilterSettings.PastSampleResetSeconds;
122  goto clamp_and_log_result;
123  }
124  else if (systemTime > PastSampleResetTime)
125  {
126  OVR_DEBUG_LOG(("SensorTimeFilter - Filtering reset due to samples in the past!\n"));
127  initClockSampling(sampleDeviceTime, clockDelta);
128  // Fall through to below, to ' PastSampleResetTime = 0.0; '
129  }
130  else
131  {
132  goto clamp_and_log_result;
133  }
134  }
135 
136  // Most common case: Record window sample.
137  else if ( (deviceTimeDelta < 1.0f) &&
138  ( (sampleDeviceTime < MinWindowLastTime) ||
139  (MinWindowSamples < FilterSettings.MinSamples) ) )
140  {
141  // Pick minimum ClockDelta sample.
142  if (clockDelta < MinWindowClockDelta)
143  MinWindowClockDelta = clockDelta;
144  MinWindowSamples++;
145  }
146  else
147  {
148  processFinishedMinWindow(sampleDeviceTime, clockDelta);
149  }
150 
151  PastSampleResetTime = 0.0;
152  }
153  else
154  {
155  initClockSampling(sampleDeviceTime, clockDelta);
156  }
157 
158 
159  // Clock adjustment for drift.
160  ClockDelta += ClockDeltaDriftPerSecond * deviceTimeDelta;
161 
162  // ClockDelta "nudging" towards last known MinWindowClockDelta.
163  if (ClockDeltaCorrectSecondsLeft > 0.000001)
164  {
165  double correctTimeDelta = deviceTimeDelta;
166  if (deviceTimeDelta > ClockDeltaCorrectSecondsLeft)
167  correctTimeDelta = ClockDeltaCorrectSecondsLeft;
168  ClockDeltaCorrectSecondsLeft -= correctTimeDelta;
169 
170  ClockDelta += ClockDeltaCorrectPerSecond * correctTimeDelta;
171  }
172 
173  // Record largest device time, so we know what samples to use in accumulation
174  // of min-window in the future.
175  LastLargestDeviceTime = sampleDeviceTime;
176 
177  // Compute our resulting sample time after ClockDelta adjustment.
178  result = sampleDeviceTime + ClockDelta;
179 
180 
181 clamp_and_log_result:
182 
183  OVR_TIMEFILTER_LOG_CODE( double savedResult = result; )
184 
185  // Clamp to ensure that result >= PrevResult, or not to far in the future.
186  // Future clamp primarily happens in the very beginning if we are de-queuing
187  // system buffer full of samples.
188  if (result < prevResult)
189  {
190  result = prevResult;
191  }
192  if (result > (systemTime + FilterSettings.FutureClamp))
193  {
194  result = (systemTime + FilterSettings.FutureClamp);
195  }
196 
197  OVR_TIMEFILTER_LOG_CODE(
198 
199  // Tag lines that were outside desired range, with '<' or '>'.
200  char rangeClamp = ' ';
201  char resultDeltaFar = ' ';
202 
203  if (savedResult > (systemTime + 0.0000001))
204  rangeClamp = '>';
205  if (savedResult < prevResult)
206  rangeClamp = '<';
207 
208  // Tag any result delta outside desired threshold with a '*'.
209  if (fabs(deviceTimeDelta - (result - prevResult)) >= 0.00002)
210  resultDeltaFar = '*';
211 
212  fprintf(pTFLogFile, "Res%s = %13.7f, dt = % 8.7f, ClkD = %13.6f "
213  "sysT = %13.6f, sysDt = %f, "
214  "sysDiff = % f, devT = %11.6f, ddevT = %9.6f %c%c\n",
215  debugTag, result, result - prevResult, ClockDelta,
216  systemTime, systemTime - PrevSystemTime,
217  -(systemTime - result), // Negatives in the past, positive > now.
218  sampleDeviceTime, deviceTimeDelta, rangeClamp, resultDeltaFar);
219 
220  ) // OVR_TIMEFILTER_LOG_CODE()
221  OVR_UNUSED(debugTag);
222 
223  // Record prior values. Useful or logging and clamping.
224  PrevSystemTime = systemTime;
225 
226  return result;
227 }

247 void SensorTimeFilter::processFinishedMinWindow(double sampleDeviceTime, double clockDelta)
248 {
249  MinRecord newRec = { MinWindowClockDelta, sampleDeviceTime };
250 
251  double clockDeltaDiff = MinWindowClockDelta - ClockDelta;
252  double absClockDeltaDiff = fabs(clockDeltaDiff);
253 
254 
255  // Abrupt change causes Reset of minClockDelta collection.
256  // > 8 ms would a Large jump in a minimum sample, as those are usually stable.
257  // > 1 second intantaneous jump would land us here as well, as that would imply
258  // device being suspended, clock wrap or some other unexpected issue.
259  if ((absClockDeltaDiff > 0.008) ||
260  ((sampleDeviceTime - LastLargestDeviceTime) >= 1.0))
261  {
262  OVR_TIMEFILTER_LOG_CODE(
263  fprintf(pTFLogFile,
264  "\nMinWindow Finished: %d Samples, MinWindowClockDelta=%f, MW-CD=%f,"
265  " ** ClockDelta Reset **\n\n",
266  MinWindowSamples, MinWindowClockDelta, MinWindowClockDelta-ClockDelta);
267  )
268 
269  // Use old collected ClockDeltaDriftPerSecond drift value
270  // up to 1 minute until we collect better samples.
271  if (!MinRecords.IsEmpty())
272  {
273  OldClockDeltaDriftExpire = MinRecords.GetNewest().LastSampleDeviceTime -
274  MinRecords.GetOldest().LastSampleDeviceTime;
275  if (OldClockDeltaDriftExpire > 60.0)
276  OldClockDeltaDriftExpire = 60.0;
277  OldClockDeltaDriftExpire += sampleDeviceTime;
278  }
279 
280  // Jump to new ClockDelta value.
281  if ((sampleDeviceTime - LastLargestDeviceTime) > 1.0)
282  ClockDelta = clockDelta;
283  else
284  ClockDelta = MinWindowClockDelta;
285 
286  ClockDeltaCorrectSecondsLeft = 0;
287  ClockDeltaCorrectPerSecond = 0;
288 
289  // Reset buffers, we'll be collecting a new MinWindow.
290  MinRecords.Reset();
291  MinWindowsCollected = 0;
292  MinWindowDuration = 0.25;
293  MinWindowSamples = 0;
294  }
295  else
296  {
297  OVR_ASSERT(MinWindowSamples >= FilterSettings.MinSamples);
298 
299  double timeElapsed = 0;
300 
301  // If we have older values, use them to update clock drift in
302  // ClockDeltaDriftPerSecond
303  if (!MinRecords.IsEmpty() && (sampleDeviceTime > OldClockDeltaDriftExpire))
304  {
305  MinRecord rec = MinRecords.GetOldest();
306 
307  // Compute clock rate of drift.
308  timeElapsed = sampleDeviceTime - rec.LastSampleDeviceTime;
309 
310  // Check for divide by zero shouldn't be necessary here, but just be be safe...
311  if (timeElapsed > 0.000001)
312  {
313  ClockDeltaDriftPerSecond = (MinWindowClockDelta - rec.MinClockDelta) / timeElapsed;
314  ClockDeltaDriftPerSecond = clampRate(ClockDeltaDriftPerSecond,
315  FilterSettings.MaxChangeRate);
316  }
317  else
318  {
319  ClockDeltaDriftPerSecond = 0.0;
320  }
321  }
322 
323  MinRecords.AddRecord(newRec);
324 
325 
326  // Catchup correction nudges ClockDelta towards MinWindowClockDelta.
327  // These are needed because clock drift correction alone is not enough
328  // for past accumulated error/high-granularity clock delta changes.
329  // The further away we are, the stronger correction we apply.
330  // Correction has timeout, as we don't want it to overshoot in case
331  // of a large delay between samples.
332 
333  if (absClockDeltaDiff >= 0.00125)
334  {
335  // Correct large discrepancy immediately.
336  if (absClockDeltaDiff > 0.00175)
337  {
338  if (clockDeltaDiff > 0)
339  ClockDelta += (clockDeltaDiff - 0.00175);
340  else
341  ClockDelta += (clockDeltaDiff + 0.00175);
342 
343  clockDeltaDiff = MinWindowClockDelta - ClockDelta;
344  }
345 
346  ClockDeltaCorrectPerSecond = clockDeltaDiff;
347  ClockDeltaCorrectSecondsLeft = 1.0;
348  }
349  else if (absClockDeltaDiff > 0.0005)
350  {
351  ClockDeltaCorrectPerSecond = clockDeltaDiff / 8.0;
352  ClockDeltaCorrectSecondsLeft = 8.0;
353  }
354  else
355  {
356  ClockDeltaCorrectPerSecond = clockDeltaDiff / 15.0;
357  ClockDeltaCorrectSecondsLeft = 15.0;
358  }
359 
360  ClockDeltaCorrectPerSecond = clampRate(ClockDeltaCorrectPerSecond,
361  FilterSettings.MaxCorrectRate);
362 
363  OVR_TIMEFILTER_LOG_CODE(
364  fprintf(pTFLogFile,
365  "\nMinWindow Finished: %d Samples, MinWindowClockDelta=%f, MW-CD=%f,"
366  " tileElapsed=%f, ClockChange=%f, ClockCorrect=%f\n\n",
367  MinWindowSamples, MinWindowClockDelta, MinWindowClockDelta-ClockDelta,
368  timeElapsed, ClockDeltaDriftPerSecond, ClockDeltaCorrectPerSecond);
369  )
370  }
371 
372  // New MinClockDelta collection window.
373  // Switch to longer duration after first few windows.
374  MinWindowsCollected ++;
375  if (MinWindowsCollected > 5)
376  MinWindowDuration = 0.5;
377 
378  MinWindowClockDelta = clockDelta;
379  MinWindowLastTime = sampleDeviceTime + MinWindowDuration;
380  MinWindowSamples = 0;
381 }
